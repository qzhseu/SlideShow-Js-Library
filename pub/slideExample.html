Slide.js

Slide.js enables you to create amazing slides using HTML. This demo will show you some examples of features.

Home Page: Home

Edit slides

Try to press "SHIFT" and click the header at the same time.

After the selected box appears, you can drag it to any position, adjust the size by dragging the circles at the four corners, and click the circle in the top middle to rotate it. Press "SHIFT" and click other element to switch element. You can press o to exit the edit mode.

                            function ExampleOne() {
                                const [time, setTime] = useState(0);
                                return (
                                    <h2> Time: {time} </h2>
                                );
                            }

                            function ExampleTwo() {
                                const [year, setYear] = useState(1990);

                                return (
                                    <h2> Year: {year} </h2>
                                    <button onClick={() => setYear(year + 1)}>
                                        Add Year
                                    </button>
                                );
                            }
